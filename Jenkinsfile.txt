pipeline {
    agent any

    stages {
        stage('Build') {
            steps {
                bat 'mvn clean package'
            }
        }
        stage('Docker Build & Push') {
            steps {
                withCredentials([string(credentialsId: 'dockerhub-pass', variable: 'DOCKER_PASS')]) {
                    bat 'docker build -t yourdockerhub/demo-java-cicd .'
                    bat 'echo $DOCKER_PASS | docker login -u yourdockerhub --password-stdin'
                    bat 'docker push yourdockerhub/demo-java-cicd'
                }
            }
        }
    }
}
